\part
The ellipse in the graph is given by the equation
\begin{align*}
	4x^2 + 9y^2 = 36.
\end{align*}
Use implicit differentiation to find a formula for the derivative $\dfrac{dy}{dx}$.

\begin{asy}
size(5cm);
import graph;
for (int i = -5; i <= 5; ++i)
	{
    draw((-5,i)--(5,i), gray);
    draw((i,-5)--(i,5), gray);
    }
draw(Label("$x$", EndPoint, E), (-6,0)--(6,0), Arrows);
draw(Label("$y$", EndPoint, N), (0,-6)--(0,6), Arrows);
pair F(real t) { 
	return (t, (1/3.0) * sqrt(36 - 4*t^2));
}
path g = graph(F, -3.0, 3.0, operator ..);
draw(g);
pair G(real t) { 
	return (t, (-1/3.0) * sqrt(36 - 4*t^2));
}
path h = graph(G, -3.0, 3.0, operator ..);
draw(h);
\end{asy}

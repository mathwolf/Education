\noindent {\sc {\bf {\Large Neighborhoods and limits}}}
\bigskip
\bigskip

\begin{enumerate}
    
	\item Let $f(x) = \dfrac{1}{2} x - 3$, and consider $\displaystyle \lim_{x \to 4} f(x) = -1$.  Find a neighborhood of $x = 4$ that 
    	guarantees $y$ is in the neighborhood $(-2,0)$.
    \vspace{\stretch{2}}

	\item Let $g(x) = 3x + \dfrac{1}{2}$, and consider $\displaystyle \lim_{x \to -2} f(x) = - \dfrac{11}{2}$.  Find a constant $\varepsilon$ so $\left| x - (-2) \right| < \varepsilon$  guarantees that $\left| f(x) - \left( - \dfrac{11}{2} \right) \right| < \dfrac{1}{2}$.     \vspace{\stretch{3}}

\clearpage
	\item Challenge: let $h(x) = x^2$, and consider $\displaystyle \lim_{x \to 3} f(x) = 9$.  Find a constant $\varepsilon$ so $\left| x - 3 \right| < \varepsilon$  guarantees that $\left| f(x) - 9 \right| < \dfrac{1}{2}$.    

\end{enumerate}

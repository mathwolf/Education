\question[10]
A experimental aircraft is launched at time $t=0$.  Its height above the ground is given by the function \begin{equation*}
	y(t) = -5t^2 + 90t + 250 ,
\end{equation*}
where $y$ is height (in meters) and $t$ is time (in seconds).
\begin{parts}
\part
How high is the aircraft at the time it is launched?
\vspace{\stretch{2}}

\part
How high is it after 3 seconds?
\vspace{\stretch{2}}

\part
What is its maximum height?
\vspace{\stretch{2}}

\part
At what time does the aircraft strike the ground?
\vspace{\stretch{3}}

\end{parts}

\end{questions}

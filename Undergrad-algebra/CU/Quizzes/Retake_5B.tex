\begin{multicols}{2}
\question[3]
$f(x) = x^{\nicefrac{1}{2}}$

\vspace{\stretch{1}}

\begin{asy}
size(7cm);
import graph;
for (int i = -5; i <= 5; ++i)
	{
    draw((-5,i)--(5,i), gray);
    draw((i,-5)--(i,5), gray);
    }
draw(Label("$x$", EndPoint, E), (-6,0)--(6,0), Arrows);
draw(Label("$y$", EndPoint, N), (0,-6)--(0,6), Arrows);
\end{asy}

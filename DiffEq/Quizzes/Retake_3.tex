\question Consider the differential equation
\begin{align*}
	\dfrac{dy}{dx} = x(y+1)^3.
\end{align*}

\begin{parts}
\part[2]
Give the order of this differential equation.
\vspace{\stretch{1}}

\part[2]
Describe the equation as linear or nonlinear.
\vspace{\stretch{1}}

\part[10]
Find all solutions of the differential equation.
\vspace{\stretch{9}}

\part[3]
Find the solution corresponding to the initial condition $y(0) = -1$.
\vspace{\stretch{2}}

\part[3]
Find the solution corresponding to the initial condition $y(0) = 1$.
\vspace{\stretch{3}}

\end{parts}

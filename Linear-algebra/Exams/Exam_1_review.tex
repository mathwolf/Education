\question
Solve the system of linear equations.  Use a parameter to stand in for any free variable in the solution.
\begin{parts}
\part
\begin{align*}
	2y + 3z & = 3 \\
    x + y + z & = 4 \\
    4x + 8y - 3z & = 35
\end{align*}
\part
\begin{align*}
	x_1 - 3x_2 + 2x_3 - x_4 + 2x_5 & = 2 \\
	3x_1 - 9x_2 + 7x_3 - x_4 + 3x_5 & = 7 \\
	2x_1 - 6x_2 + 7x_3 + 4x_4 - 5x_5 & = 7 \\
\end{align*}
\part
\begin{align*}
	x_1 + 2x_2 - 3x_3 + 4x_4 & = 2 \\
	2x_1 + 5x_2 - 2x_3 + x_4 & = 1 \\
	5x_1 + 12x_2 - 7x_3 + 6x_4 & = 3 \\
\end{align*}
\end{parts}

\question
Define the matrices $A = \left[ \begin{array}{cc}
1 & 3 \\ 
2 & -1 
\end{array} \right],
B = 
\left[ \begin{array}{ccc}
2 & 0 & 4 \\
3 & -2 & 6
\end{array} \right]$.
\begin{parts}
\part
Find $AB$.
\part
Find $BA$.
\end{parts}

\question
Find the inverse of the matrix if it exists.
\smallskip
\begin{parts}
\part
$A = \left[ \begin{array}{cc}
2 & -3 \\ 
1 & 3 
\end{array} \right]$

\part
$A = \left[ \begin{array}{cc}
-2 & 6 \\ 
3 & -9 
\end{array} \right]$
\end{parts}

\end{questions}

\begin{questions}
\setcounter{question}{2}
\question
For the system
\begin{align*}
	x + 2y + z & = 3 \\
    ay + 5z & = 10 \\
    2x + 7y + az & = b,
\end{align*}
\begin{parts}
\part
Find the values of the pair $(a,b)$ for which the system has a unique solution.
\part
Find the values of the pair $(a,b)$ for which the system has no solution.
\part
Find the values of the pair $(a,b)$ for which the system has infinite solutions.
\end{parts}

\question
Describe the shape made by span of the vector 
$\vec{v} = \left[ \begin{array}{c}
1 \\ 0 \\ -5
\end{array} \right]$.

\question
Does the vector 
$\vec{v} = \left[ \begin{array}{c}
3 \\ 0 \\ -4
\end{array} \right]$
lie in
$\textnormal{Span} \left\lbrace \left[ \begin{array}{c}
1 \\ 2 \\ 0
\end{array} \right], 
\left[ \begin{array}{c}
-1 \\ 1 \\ 2
\end{array} \right]
\right\rbrace$?

\question
Determine whether the given vectors form a linearly independent set.
\begin{parts}
\part
$\vec{u} = \left[ \begin{array}{c}
3 \\ -5
\end{array} \right]$,
$\vec{v} = \left[ \begin{array}{c}
1 \\ 2 
\end{array} \right]$

\part
$\vec{u} = \left[ \begin{array}{c}
1 \\ -3
\end{array} \right]$,
$\vec{v} = \left[ \begin{array}{c}
-2 \\ 6 
\end{array} \right]$

\part
$\vec{u} = \left[ \begin{array}{c}
1 \\ 1 \\ 2
\end{array} \right]$,
$\vec{v} = \left[ \begin{array}{c}
2 \\ 3 \\ 1 
\end{array} \right]$
$\vec{w} = \left[ \begin{array}{c}
4 \\ 4 \\ 5 
\end{array} \right]$

\end{parts}

\question
Consider the following linear transformation $T$ that acts on $\mathbf{R}^2$.  First the plane is rotated $225 \degree$ in a counterclockwise direction.  Next, the plane is reflected across the y-axis.  Finally, the plane is stretched by a factor of 3 in the vertical direction.
\begin{parts}
\part
Draw two pictures that show the locations of the unit coordinate vectors $\vec{e_1}$ and $\vec{e_2}$ before and after the transformation $T$.

\part
Write the matrix for this transformation. 

\end{parts}

\question
Give one example of a linear transformation from $\mathbf{R}^2$ to $\mathbf{R}^2$ that is one-to-one.  Give an example of a transformation that is not one-to-one.  Here I am looking for a description of the geometry of a transformation and not the matrix.

\question 
Suppose that two $n \times n$ matrices $A$ and $B$ are both invertible.  Show that their product $AB$ is invertible.

\end{questions}
